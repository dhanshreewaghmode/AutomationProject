package strin;
//matix formed
public class Multidimension {
	public static void main(String[] args) {
		int a[] []= new int [2][2];
		a[0][0]=70;
		a[0][1]=50;
		a[1][0]=40;//70 50  
		            //40 55
		a[1][1]=55;
		
		
		for(int i=0;i<=1;i++) {
			
			for (int j=0;j<=1;j++) {
				System.out.print(a[i][j]);
				System.out.print(" ");
			}
			System.out.println(" ");
		}
		
		
	}

}
/*output         70 50  
                  40 55*/
